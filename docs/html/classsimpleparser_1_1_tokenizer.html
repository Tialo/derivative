Пример создания и использования

#include <iostream>
#include <parser/parser.cpp>
int main(){
simpleparser::Tokenizer tz;
auto x = tz.Tokenize("(x + 22) * cos(x / 2)");
for (auto s : x) {
std::cout << s.text << " | ";
}
std::cout << "\n\n";
}
